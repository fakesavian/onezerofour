```jsx
import React, { useState, useEffect, useRef } from 'react';
import { motion } from 'framer-motion';
import storyContent from '../data/storyContent.json';
import './Terminal.css';

const Terminal = () => {
  const [text, setText] = useState('');
  const [currentArchitect, setCurrentArchitect] = useState(null);
  const terminalRef = useRef(null);

  const typeWriter = (message, speed = 50) => {
    let i = 0;
    const typing = setInterval(() => {
      if (i < message.length) {
        setText(prevText => prevText + message.charAt(i));
        i++;
      } else {
        clearInterval(typing);
      }
    }, speed);
  };

  useEffect(() => {
    typeWriter(storyContent.initialMessage);
  }, []);

  const handleArchitectSelect = (architect) => {
    setCurrentArchitect(architect);
    typeWriter(architect.dialogue);
  };

  return (
    <motion.div 
      ref={terminalRef}
      className="terminal"
      initial={{ opacity: 0 }}
      animate={{ opacity: 1 }}
      transition={{ duration: 1 }}
    >
      <div className="terminal-screen">
        <pre>{text}</pre>
        <div className="architect-selection">
          {storyContent.architects.map(architect => (
            <button 
              key={architect.id}
              onClick={() => handleArchitectSelect(architect)}
            >
              {architect.id}
            </button>
          ))}
        </div>
      </div>
    </motion.div>
  );
};

export default Terminal;
```
